# L1-006 连续因子
```cpp
#include <bits/stdc++.h>
using i64 = long long;
int main()
{
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n;
    std::cin >> n;

    int first = 0, len = 0;
    for (int i = 2; i <= n / i; i++)
    {
        int tmp = 1, j;
        for (j = i; j <= n / i; j++)
        {
            tmp *= j;
            if (n % tmp)
                break;
        }

        if (len < j - i)
            len = j - i, first = i;
    }

    if (!first)
        std::cout << 1 << '\n' << n << '\n';
    else
    {
        std::cout << len << '\n' << first++;
        for (int i = 2; i <= len; i++)
            std::cout << "*" << first++;
    }
    return 0;
}
```

```cpp
    for (int i = 0; i < n; i++)
    {
        int a, b;
        scanf("%d/%d", &a, &b);
        sum_a *= b;
        sum_a += a * sum_b;
        sum_b *= b;
        int t = std::gcd(sum_a, sum_b);
        sum_a /= t, sum_b /= t;
    }

    if (sum_a % sum_b == 0)
        std::cout << sum_a / sum_b;
    else if (sum_a < sum_b)
        std::cout << sum_a << '/' << sum_b;
    else
        std::cout << sum_a / sum_b << " " << sum_a % sum_b << "/" << sum_b;
    return 0;
}
```

```cpp
#include <bits/stdc++.h>

int main()
```

```cpp
    std::string a, b, t;
    std::getline(std::cin, t);

    for (int i = 0; i < t.size(); i++)
    {
        if (t[i] == ' ')
        {
            a = t.substr(0, i);

        return true;
    };

        a = "?";
    if (!islegal(b))
        b = "?";
    std::cout << a << " + " << b << " = ";
    if (a != "?" && b != "?")
        std::cout << std::stoi(a) + std::stoi(b);
    else
        std::cout << "?";
```

```cpp
struct DSU
{
	std::vector<int> f, siz;
	DSU(int n) : f(n), siz(n, 1)
	{
		std::iota(f.begin(), f.end(), 0);
	}
	int leader(int x)
	{
		return x == f[x] ? x : f[x] = leader(f[x]);
	}
	bool same(int x, int y)
	{
		return leader(x) == leader(y);
	}
	void merge(int x, int y)
	{
		x = leader(x), y = leader(y);
		int mx = std::max(x, y), mn = std::min(x, y); // 保证家庭编号最小
		if (x != y)
		{
			siz[mn] += siz[mx];
			f[mx] = mn;
		}
	}
	int size(int x)
	{
		return siz[leader(x)];
	}
};

	int id;		 // 家庭成员的最小编号
	int all_peo; // 家庭人口数
```

```kotlin
#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll a[200005];

int main()
{
    ll n,v;
    scanf("%lld%lld",&n,&v);
    for(int i=0;i<n;i++)
        scanf("%lld",&a[i]);    
    ll ans=0;
    sort(a,a+n);
    for(int i=0;i<n;i++){
        ll x = a[i], y = v;
        //寻找z满足x+y>z , x+z > y ,y + z > x ,即 z<x+y 且 z > max(y-x,x-y) == abs(x-y)
        //每次搜索区间以当前区间为起点 避免重复
        //返回第一个>=x+y的值 则[i,r]为<x+y的z
        int r = lower_bound(a+1+i,a+n,x+y) - a;

        //返回第一个>abs(x-y)的值的下标 则[l,n]为>abs(x+y)的z
        ll l = upper_bound(a+1+i,a+n,abs(x-y)) -a;
        ans += r -l;
    }
    cout<<ans;
    return 0;
}
```
